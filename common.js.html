/**
 * Remove falsy values
 * @param {Object} obj object to process
 */
export function removeFalsy(obj) {
  if (typeof obj !== "object" || Array.isArray(obj)) return obj;
  for (var i in obj) {
    if (!obj[i]) {
      delete obj[i];
    } else {
      obj[i] = removeFalsy(obj[i]);
      if (!obj[i]) {
        delete obj[i];
      }
    }
  }
  return Object.keys(obj).length ? obj : null;
}
